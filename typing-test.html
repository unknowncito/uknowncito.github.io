Typing Test

A typing speed and accuracy test application.

Source Code

from cmu_graphics import *
import random

def onAppStart(app):
    app.word = getNewWord()
    app.letterIndex = 0
    app.typos = 0

def getNewWord():
    words = ['apple', 'banana', 'cherry', 'date', 'elderberry',
             'fig', 'grape', 'honeydew', 'kiwi', 'lemon']
    return random.choice(words)

def redrawAll(app):
    drawRect(0, 0, 400, 400, fill='white')
    drawLabel('Typing Test', 200, 50, size=30, bold=True)
    drawLabel(f'Typos: {app.typos}', 200, 100, size=20)
    drawLabel('Press Escape for new word', 200, 350, size=16)
    
    # Draw the word with colored boxes
    startX = 200 - len(app.word) * 15
    for i in range(len(app.word)):
        letter = app.word[i]
        x = startX + i * 30
        y = 200
        
        if i < app.letterIndex:
            # Already typed - green
            fill = 'lightGreen'
        elif i == app.letterIndex:
            # Current letter - white
            fill = 'white'
        else:
            # Not yet typed - pink
            fill = 'pink'
        
        drawRect(x, y, 25, 40, fill=fill, border='black')
        drawLabel(letter, x + 12.5, y + 20, size=20, bold=True)

def onKeyPress(app, key):
    if key == 'escape':
        app.word = getNewWord()
        app.letterIndex = 0
    elif app.letterIndex < len(app.word):
        if key == app.word[app.letterIndex]:
            app.letterIndex += 1
            if app.letterIndex == len(app.word):
                # Word complete!
                app.word = getNewWord()
                app.letterIndex = 0
        else:
            app.typos += 1

def main():
    runApp()

main()
