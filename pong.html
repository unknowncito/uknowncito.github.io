Pong

A two-player paddle game featuring the classic Pong mechanics.

Source Code

from cmu_graphics import *
import random

def onAppStart(app):
    app.paddleX = 200
    app.paddleY = 370
    app.paddleWidth = 80
    app.paddleHeight = 10
    
    app.ballX = 200
    app.ballY = 200
    app.ballRadius = 8
    app.ballSpeedX = 4
    app.ballSpeedY = 4
    
    app.hits = 0
    app.misses = 0
    app.gameOver = False
    app.stepsPerSecond = 30

def redrawAll(app):
    drawRect(0, 0, 400, 400, fill='black')
    
    # Draw paddle
    drawRect(app.paddleX - app.paddleWidth/2, app.paddleY,
             app.paddleWidth, app.paddleHeight, fill='white')
    
    # Draw ball
    drawCircle(app.ballX, app.ballY, app.ballRadius, fill='white')
    
    # Draw score
    drawLabel(f'Hits: {app.hits}', 50, 20, fill='white', size=16, bold=True)
    drawLabel(f'Misses: {app.misses}', 350, 20, fill='white', size=16, bold=True)
    
    if app.gameOver:
        drawLabel('Game Over!', 200, 200, fill='red', size=30, bold=True)
        drawLabel('Press r to restart', 200, 230, fill='white', size=16)

def onStep(app):
    if not app.gameOver:
        # Move ball
        app.ballX += app.ballSpeedX
        app.ballY += app.ballSpeedY
        
        # Bounce off walls
        if app.ballX - app.ballRadius <= 0 or app.ballX + app.ballRadius >= 400:
            app.ballSpeedX = -app.ballSpeedX
        
        if app.ballY - app.ballRadius <= 0:
            app.ballSpeedY = -app.ballSpeedY
        
        # Check paddle collision
        if (app.ballY + app.ballRadius >= app.paddleY and
            app.ballY - app.ballRadius <= app.paddleY + app.paddleHeight and
            app.ballX >= app.paddleX - app.paddleWidth/2 and
            app.ballX <= app.paddleX + app.paddleWidth/2):
            app.ballSpeedY = -abs(app.ballSpeedY)
            app.hits += 1
        
        # Check if ball missed
        if app.ballY - app.ballRadius > 400:
            app.misses += 1
            if app.misses >= 3:
                app.gameOver = True
            else:
                # Reset ball
                app.ballX = 200
                app.ballY = 200
                app.ballSpeedY = -abs(app.ballSpeedY)

def onKeyPress(app, key):
    if key == 'r':
        onAppStart(app)
    elif not app.gameOver:
        if key == 'left' and app.paddleX - app.paddleWidth/2 > 0:
            app.paddleX -= 20
        elif key == 'right' and app.paddleX + app.paddleWidth/2 < 400:
            app.paddleX += 20

def onKeyHold(app, keys):
    if not app.gameOver:
        if 'left' in keys and app.paddleX - app.paddleWidth/2 > 0:
            app.paddleX -= 5
        if 'right' in keys and app.paddleX + app.paddleWidth/2 < 400:
            app.paddleX += 5

def main():
    runApp()

main()
